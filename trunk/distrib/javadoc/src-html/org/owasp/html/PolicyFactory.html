029    package org.owasp.html;
030    
031    import java.util.Map;
032    
033    import javax.annotation.Nullable;
034    import javax.annotation.concurrent.Immutable;
035    import javax.annotation.concurrent.ThreadSafe;
036    
037    import com.google.common.base.Function;
038    import com.google.common.collect.ImmutableMap;
039    
040    /**
041     * A factory that can be used to link a sanitizer to an output receiver and that
042     * provides a convenient <code>{@link PolicyFactory#sanitize sanitize}</code>
043     * method and a <code>{@link PolicyFactory#and and}</code> method to compose
044     * policies.
045     *
046     * @author Mike Samuel <mikesamuel@gmail.com>
047     */
048    @ThreadSafe
049    @Immutable
050    @TCB
051    public final class PolicyFactory
052        implements Function<HtmlStreamEventReceiver, HtmlSanitizer.Policy> {
053    
054      private final ImmutableMap<String, ElementAndAttributePolicies> policies;
055      private final boolean allowStyling;
056    
057      PolicyFactory(ImmutableMap<String, ElementAndAttributePolicies> policies,
058              boolean allowStyling) {
059        this.policies = policies;
060        this.allowStyling = allowStyling;
061      }
062    
063      /** Produces a sanitizer that emits tokens to out. */
064      public HtmlSanitizer.Policy apply(HtmlStreamEventReceiver out) {
065        if (allowStyling) {
066          return new StylingPolicy(out, policies);
067        } else {
068          return new ElementAndAttributePolicyBasedSanitizerPolicy(
069              out, policies);
070        }
071      }
072    
073      /** A convenience function that sanitizes a string of HTML. */
074      public String sanitize(@Nullable String html) {
075        if (html == null) { return ""; }
076        StringBuilder out = new StringBuilder(html.length());
077        HtmlSanitizer.sanitize(
078            html, apply(HtmlStreamRenderer.create(out, Handler.DO_NOTHING)));
079        return out.toString();
080      }
081    
082      /**
083       * Produces a factory that allows the union of the grants, and intersects
084       * policies where they overlap on a particular granted attribute or element
085       * name.
086       */
087      public PolicyFactory and(PolicyFactory f) {
088        ImmutableMap.Builder<String, ElementAndAttributePolicies> b
089            = ImmutableMap.builder();
090        for (Map.Entry<String, ElementAndAttributePolicies> e
091            : policies.entrySet()) {
092          String elName = e.getKey();
093          ElementAndAttributePolicies p = e.getValue();
094          ElementAndAttributePolicies q = f.policies.get(elName);
095          if (q != null) {
096            p = p.and(q);
097          }
098          b.put(elName, p);
099        }
100        for (Map.Entry<String, ElementAndAttributePolicies> e
101            : f.policies.entrySet()) {
102          String elName = e.getKey();
103          if (!policies.containsKey(elName)) {
104            b.put(elName, e.getValue());
105          }
106        }
107        return new PolicyFactory(b.build(), allowStyling || f.allowStyling);
108      }
109    }
